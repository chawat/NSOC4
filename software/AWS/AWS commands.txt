//Enable password authentication 
vi etc/ssh/sshd_config
	PasswordAuth yes

//Create a non-root user with sudo access
adduser username
usermod -aG sudo username
sudo reboot
//install and setup apache
sudo apt-get install apache2 apache2-doc apache2-utils
sudo a2dissite 000-default.conf

//create a virtual host
vi /etc/apache2/ssh/sites-available/nsoc.tk.conf

//The file's content
<VirtualHost *:80>
     ServerAdmin webmaster@nsoc.tk
     ServerName nsoc.tk
     ServerAlias www.nsoc.tk
     DocumentRoot /var/www/nsoc.tk/public_html/
     ErrorLog /var/www/nsoc.tk/logs/error.log
     CustomLog /var/www/nsoc.tk/logs/access.log combined
</VirtualHost>

sudo mkdir -p /var/www/nsoc.tk/public_html
sudo mkdir /var/www/nsoc.tk/logs

sudo a2ensite nsoc.tk.conf
	
sudo service apache2 restart

cd var/www/nsoc.tk/public_html/

//git clone
git init
git remote add origin https://github.com/nisbweb/awsproject.git
git pull origin master

 sudo apt-get install software-properties-common
